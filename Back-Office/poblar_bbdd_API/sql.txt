CREATE TABLE STATS (
    id INT PRIMARY KEY,
    player_id INT,
    game_id INT,
    team_id INT,
    min INT,
    fgm INT,
    fga INT,
    fg_pct DECIMAL(5, 4),
    fg3m INT,
    fg3a INT,
    fg3_pct DECIMAL(5, 4),
    ftm INT,
    fta INT,
    ft_pct DECIMAL(5, 4),
    oreb INT,
    dreb INT,
    reb INT,
    ast INT,
    stl INT,
    blk INT,
    turnover INT,
    pf INT,
    pts INT,
    FOREIGN KEY (player_id) REFERENCES PLAYERS(id),
    FOREIGN KEY (game_id) REFERENCES GAMES(id),
    FOREIGN KEY (team_id) REFERENCES TEAMS(id)
);
